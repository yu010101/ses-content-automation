```python
# Few-shot プロンプトの例
def create_few_shot_prompt(task_description, examples, new_input):
    """
    効果的なFew-shotプロンプトを生成する関数
    """
    prompt = f"""タスク: {task_description}

例:
"""
    
    for i, example in enumerate(examples, 1):
        prompt += f"""
例{i}:
入力: {example['input']}
出力: {example['output']}
"""
    
    prompt += f"""

新しいタスク:
入力: {new_input}
出力:"""
    
    return prompt

# 使用例
```

### Phase 2: RAGシステムの構築

#### 2.1 ベーシックなRAGパイプライン

```python
import openai
from sentence_transformers import SentenceTransformer
import faiss
import numpy as np
from typing import List, Dict

class SimpleRAGSystem:
    def __init__(self, openai_api_key: str):
        self.client = openai.OpenAI(api_key=openai_api_key)
        self.encoder = SentenceTransformer('all-MiniLM-L6-v2')
        self.index = None
        self.documents = []
        
    def add_documents(self, documents: List[str]):
        """
        ドキュメントをベクトル化してインデックスに追加
        """
        self.documents.extend(documents)
        
        # エンベディング生成
        embeddings = self.encoder.encode(documents)
        
        # FAISSインデックスの初期化
        if self.index is None:
            dimension = embeddings.shape[1]
            self.index = faiss.IndexFlatIP(dimension)
            
        # 正規化
        faiss.normalize_L2(embeddings)
        
        # インデックスに追加
        self.index.add(embeddings.astype('float32'))
        
    def retrieve_documents(self, query: str, top_k: int = 3) -> List[str]:
        """
        クエリに関連するドキュメントを検索
        """
        query_embedding = self.encoder.encode([query])
        faiss.normalize_L2(query_embedding)
        
        # 類似検索
        scores, indices = self.index.search(query_embedding.astype('float32'), top_k)
        
        # 関連ドキュメントを返す
        retrieved_docs = [self.documents[idx] for idx in indices[0]]
        return retrieved_docs
        
    def generate_answer(self, query: str) -> str:
        """
        RAGを使用して回答を生成
        """
        # 関連ドキュメントを検索
        context_docs = self.retrieve_documents(query)
        
        # コンテキストを結合
        context = "\n\n".join(context_docs)
        
        # プロンプト構築
        prompt = f"""
以下のコンテキスト情報を基に、質問に答えてください。

コンテキスト:
{context}

質問: {query}

回答:"""
        
        # LLMで回答生成
        response = self.client.chat.completions.create(
            model="gpt-3.5-turbo",
            messages=[{"role": "user", "content": prompt}],
            temperature=0.1
        )
        
        return response.choices[0].message.content

# 使用例
rag_system = SimpleRAGSystem("your-openai-key")

# 技術ドキュメントを追加
technical_docs = [
    "Pythonは動的型付け言語で、シンプルで読みやすい構文が特徴です。",
    "機械学習ではscikit-learn、pandas、numpyが主要なライブラリです。",
    "FastAPIはPythonの高性能WebフレームワークでAPI開発に適しています。",
    "Dockerはコンテナ仮想化技術でアプリケーションの配布を簡素化します。"
]
```

#### 2.2 改良版RAG：リランキング機能付き

```python
from sentence_transformers import CrossEncoder
from typing import List, Tuple

class AdvancedRAGSystem(SimpleRAGSystem):
    def __init__(self, openai_api_key: str):
        super().__init__(openai_api_key)
        # クロスエンコーダーでリランキング
        self.reranker = CrossEncoder('cross-encoder/ms-marco-MiniLM-L-6-v2')
        
    def rerank_documents(self, query: str, documents: List[str], top_k: int = 3) -> List[str]:
        """
        クロスエンコーダーを使用してドキュメントをリランキング
        """
        if not documents:
            return []
            
        # クエリとドキュメントのペアを作成
        pairs = [(query, doc) for doc in documents]
        
        # 関連度スコアを計算
        scores = self.reranker.predict(pairs)
        
        # スコアでソート
        doc_scores = list(zip(documents, scores))
        doc_scores.sort(key=lambda x: x[1], reverse=True)
        
        # 上位k個を返す
        return [doc for doc, _ in doc_scores[:top_k]]
    
    def generate_answer_with_reranking(self, query: str, initial_k: int = 10, final_k: int = 3) -> str:
        """
        リランキングを使用した高精度な回答生成
        """
        # 初期検索（多めに取得）
        initial_docs = self.retrieve_documents(query, top_k=initial_k)
        
        # リランキング
        final_docs = self.rerank_documents(query, initial_docs, top_k=final_k)
        
        # コンテキスト構築
        context = "\n\n".join(final_docs)
        
        # プロンプト構築（より詳細な指示）
        prompt = f"""
以下のコンテキスト情報を基に、質問に正確かつ詳細に答えてください。
コンテキストに含まれない情報については推測せず、「コンテキストに含まれていません」と回答してください。

コンテキスト:
{context}

質問: {query}

回答（根拠も含めて）:"""
        
        response = self.client.chat.completions.create(
            model="gpt-4",
            messages=[{"role": "user", "content": prompt}],
            temperature=0.1
        )
        
        return response.choices[0].message.content
```

### Phase 3: LLM API の効果的な活用

#### 3.1 API コスト最適化戦略

```python
import time
from functools import wraps
from typing import Dict, Any

class CostOptimizedLLMClient:
    def __init__(self, openai_api_key: str):
        self.client = openai.OpenAI(api_key=openai_api_key)
        self.cache = {}
        self.request_count = 0
        self.total_tokens = 0
        
    def cached_completion(self, messages: list, model: str = "gpt-3.5-turbo", **kwargs):
        """
        キャッシュ機能付きのAPI呼び出し
        """
        # キャッシュキーを生成
        cache_key = hash(str(messages) + str(kwargs))
        
        if cache_key in self.cache:
            print("キャッシュヒット - API呼び出しをスキップ")
            return self.cache[cache_key]
            
        # API呼び出し
        response = self.client.chat.completions.create(
            model=model,
            messages=messages,
            **kwargs
        )
        
        # 統計更新
        self.request_count += 1
        self.total_tokens += response.usage.total_tokens
        
        # キャッシュに保存
        self.cache[cache_key] = response
        
        return response
        
    def batch_completion(self, message_batches: List[List[Dict]], model: str = "gpt-3.5-turbo"):
        """
        バッチ処理でAPI呼び出しを効率化
        """
        results = []
        
        for batch in message_batches:
            try:
                response = self.cached_completion(batch, model=model)
                results.append(response.choices[0].message.content)
                
                # レート制限対応
                time.sleep(0.1)
                
            except Exception as e:
                print(f"エラー発生: {e}")
                results.append(None)
                
        return results
        
    def get_usage_stats(self) -> Dict[str, Any]:
        """
        使用統計を取得
        """
        # 概算コスト計算（gpt-3.5-turboベース）
        estimated_cost = (self.total_tokens / 1000) * 0.002
        
        return {
            "request_count": self.request_count,
            "total_tokens": self.total_tokens,
            "estimated_cost_usd": round(estimated_cost, 4),
            "cache_size": len(self.cache)
        }

# 使用例
client = CostOptimizedLLMClient("your-api-key")

# バッチ処理例
code_review_tasks = [
    [{"role": "user", "content": "以下のPythonコードをレビューしてください: def add(a, b): return a + b"}],
    [{"role": "user", "content": "以下のPythonコードをレビューしてください: def divide(a, b): return a / b"}]
]

results = client.batch_completion(code_review_tasks)
print("使用統計:", client.get_usage_stats())
```

### Phase 4: AI自動化ツールの活用

#### 4.1 コード生成自動化システム

```typescript
// TypeScript版のAI駆動開発支援ツール
import OpenAI from 'openai';
import * as fs from 'fs';
import * as path from 'path';

interface CodeGenerationRequest {
  description: string;
  language: string;
  framework?: string;
  additionalRequirements?: string[];
}

interface GeneratedCode {
  code: string;
  explanation: string;
  testCases: string;
  documentation: string;
}

class AICodeGenerator {
  private openai: OpenAI;
  
  constructor(apiKey: string) {
    this.openai = new OpenAI({ apiKey });
  }
  
  async generateCode(request: CodeGenerationRequest): Promise<GeneratedCode> {
    const prompt = this.buildPrompt(request);
    
    const response = await this.openai.chat.completions.create({
      model: "gpt-4",
      messages: [{ role: "user", content: prompt }],
      temperature: 0.2,
    });
    
    return this.parseResponse(response.choices[0].message.content || '');
  }
  
  private buildPrompt(request: CodeGenerationRequest): string {
    return `
以下の仕様に基づいて、高品質なコードを生成してください。

## 仕様
- 言語: ${request.language}
- 説明: ${request.description}
${request.framework ? `- フレームワーク: ${request.framework}` : ''}
${request.additionalRequirements ? `- 追加要件: ${request.additionalRequirements.join(', ')}` : ''}

## 出力形式
### コード
\`\`\`${request.language}
[ここにコードを記述]
\`\`\`

### 説明
[コードの説明]

### テストケース
\`\`\`${request.language}
[テストコードを記述]
\`\`\`

### ドキュメント
[使用方法とAPI仕様]
`;
  }
  
  private parseResponse(response: string): GeneratedCode {
    // レスポンスをパースして構造化データに変換
    const codeMatch = response.match(/```[\w]*\n([\s\S]*?)```/);
    const code = codeMatch ? codeMatch[1].trim() : '';
    
    // 説明部分を抽出
    const explanationMatch = response.match(/### 説明\n([\s\S]*?)(?=###|$)/);
    const explanation = explanationMatch ? explanationMatch[1].trim() : '';
    
    // テストケース部分を抽出
    const testMatch = response.match(/### テストケース\n```[\w]*\n([\s\S]*?)```/);
    const testCases = testMatch ? testMatch[1].trim() : '';
    
    // ドキュメント部分を抽出
    const docMatch = response.match(/### ドキュメント\n([\s\S]*?)$/);
    const documentation = docMatch ? docMatch[1].trim() : '';
    
    return {
      code,
      explanation,
      testCases,
      documentation
    };
  }
  
  async generateProject(projectName: string, requests: CodeGenerationRequest[]): Promise<void> {
    const projectDir = path.join(process.cwd(), projectName);
    
    // プロジェクトディレクトリを作成
    if (!fs.existsSync(projectDir)) {
      fs.mkdirSync(projectDir, { recursive: true });
    }
    
    // 各ファイルを生成
    for (const request of requests) {
      const result = await this.generateCode(request);
      
      // ファイル名を生成
      const fileName = this.generateFileName(request);
      const filePath = path.join(projectDir, fileName);
      
      // コードファイルを保存
      fs.writeFileSync(filePath, result.code);
      
      // テストファイルを保存
      if (result.testCases) {
        const testFileName = fileName.replace(/\.(\w+)$/, '.test.$1');
        const testPath = path.join(projectDir, 'tests', testFileName);
        
        // testsディレクトリを作成
        const testsDir = path.dirname(testPath);
        if (!fs.existsSync(testsDir)) {
          fs.mkdirSync(testsDir, { recursive: true });
        }
        
        fs.writeFileSync(testPath, result.testCases);
      }
      
      // READMEに追加
      const readmePath = path.join(projectDir, 'README.md');
      const readmeContent = fs.existsSync(readmePath) ? fs.readFileSync(readmePath, 'utf8') : '# ' + projectName + '\n\n';
      const updatedReadme = readmeContent + `\n## ${fileName}\n${result.explanation}\n\n${result.documentation}\n`;
      
      fs.writeFileSync(readmePath, updatedReadme);
      
      console.log(`生成完了: ${fileName}`);
    }
    
    console.log(`プロジェクト "${projectName}" の生成が完了しました。`);
  }
  
  private generateFileName(request: CodeGenerationRequest): string {
    // 説明からファイル名を推測
    const description = request.description.toLowerCase();
    let baseName = description
      .replace(/[^a-z0-9\s]/g, '')
      .replace(/\s+/g, '_')
      .substring(0, 30);
      
    // 拡張子を追加
    const extensions: { [key: string]: string } = {
      'python': '.py',
      'typescript': '.ts',
      'javascript': '.js',
      'java': '.java',
      'csharp': '.cs'
    };
    
    return baseName + (extensions[request.language.toLowerCase()] || '.txt');
  }
}
```
